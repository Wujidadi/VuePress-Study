const vueApp = Vue.createApp({
    data() {
        return {
            subPages: [
                { href: 'sub/component/', text: '組件' },
                { href: 'sub/slot/', text: '插槽' }
            ]
        };
    }
});

vueApp.component('sub-page-list', {
    props: [ 'href', 'text' ],
    template: '<li><a v-bind:href="href" target="_blank">{{ text }}</a>'
});

const vueModel = vueApp.mount('#main-page');
